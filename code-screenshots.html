3 — Middleware JWT (Node.js / Express)

middleware/verifyToken.js
JavaScript
1const jwt = require('jsonwebtoken')
2
3const verifyToken = (req, res, next) => {
4 const authHeader = req.headers['authorization']
5 const token = authHeader && authHeader.split(' ')[1]
6
7 if (!token)
8 return res.status(401).json({ message: 'Token manquant' })
9
10 jwt.verify(token, process.env.JWT_SECRET, (err, decoded) => {
11 if (err)
12 return res.status(403).json({ message: 'Token invalide' })
13
14 req.user = decoded // { id, role, email }
15 next()
16 })
17}
18
19module.exports = { verifyToken }

5 — Axios Intercepteurs JWT (React / TypeScript)

services/api.ts — Intercepteurs Axios
TypeScript
1import axios from 'axios'
2
3const api = axios.create({
4 baseURL: import.meta.env.VITE_API_URL,
5})
6
7// Injecte le JWT dans chaque requête
8api.interceptors.request.use((config) => {
9 const token = localStorage.getItem('token')
10 if (token)
11 config.headers.Authorization = `Bearer ${token}`
12 return config
13})
14
15// Gère les 401 (token expiré) globalement
16api.interceptors.response.use(
17 (res) => res,
18 (err) => {
19 if (err.response?.status === 401)
20 window.location.replace('/login')
21 return Promise.reject(err)
22 }
23)
24
25export default api
